@meta {
	name "Custom Colors Support";
	description "Installs mods to support custom colors in resource packs";
	authors "mcvm";
}

@install {
	if modloader forge {
		require "optifine-support";
		finish;
	}
	if modloader fabriclike {
		require "colormatic";
		finish;
	}
	fail unsupported_modloader;
}
