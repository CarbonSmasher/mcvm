@meta {
	name "Emissive Blocks Support";
	description "Installs mods to support emissive blocks in resource packs";
	authors "mcvm";
}

@install {
	if modloader forge {
		require "optifine-support";
		finish;
	}
	if modloader fabriclike {
		require "continuity";
		finish;
	}
	fail unsupported_modloader;
}
