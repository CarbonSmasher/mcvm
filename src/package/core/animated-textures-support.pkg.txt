@meta {
	name "Animated Textures Support";
	description "Installs mods to support animated textures in resource packs";
	authors "CarbonSmasher";
}

@install {
	if modloader forge {
		require <"optifine">;
		finish;
	}
	if modloader fabriclike {
		require "animatica";
		finish;
	}
	fail unsupported_modloader;
}
