@meta {
	name "Connected Textures Support";
	description "Installs mods to support Connected Textures in resource packs";
	authors "CarbonSmasher";
}

@install {
	if modloader forge {
		require <"optifine">;
		finish;
	}
	if modloader fabriclike {
		require "continuity";
		finish;
	}
	fail unsupported_modloader;
}
